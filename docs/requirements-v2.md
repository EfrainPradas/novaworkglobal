**B. Match Scoring Algorithm**
```python
def calculate_match_score(user_profile, job):
    score = 0
    weights = {}

    # Skills match (40% weight)
    user_skills = set(user_profile.skills)
    required_skills = extract_skills_from_job(job.description)
    skills_overlap = len(user_skills & required_skills) / len(required_skills)
    score += skills_overlap * 40

    # Experience level match (20% weight)
    years_match = compare_experience(user_profile.years_exp, job.required_years)
    score += years_match * 20

    # Title similarity (15% weight)
    title_sim = calculate_title_similarity(user_profile.target_roles, job.title)
    score += title_sim * 15

    # Location match (10% weight)
    if job.remote or job.location == user_profile.location:
        score += 10

    # Salary match (10% weight)
    if job.salary_min >= user_profile.salary_min:
        score += 10

    # Company size preference (5% weight)
    if job.company_size in user_profile.preferred_company_sizes:
        score += 5

    return min(int(score), 100)
```
